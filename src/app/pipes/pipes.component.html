Pipes Concept in Angular

uppercase pipe :

{{ title | uppercase }}

lowercase pipe :

{{ data | lowercase }}

titlecase pipe :

{{ title | titlecase }}

slice pipe with one argument :

{{ title | slice:5}}

slice with 2 arguments :

{{ slice | slice: 5: 16}}

json pipe :

{{ student | json}}

date pipe :

{{ todaydate }}
{{ todaydate | date:'d/M/y'}}
{{ todaydate | date:'MM/DD/YYYY'}} {{ todaydate | date:'short'}}
{{ tadaydate | date:'shortDate'}}
{{ todaydate | date:'shortTime'}}
{{ todaydate | date:'longTime'}}
{{ todaydate | date:'fullTime' }}
{{ todaydate | date:'longDate'}}
{{ todaydate | date:'long'}}
{{ todaydate | date:'fullDate' | uppercase}}
{{ todaydate | date:'format'}}

number pipe :

{{ 5.34567 | number:'1.2-3'}}
{{ 5.34567 | number: '2.6-7'}}
{{ 5.34567 | number: '3.2-3'}}

percentage pipe:

{{ 0.25 | percent}}
{{0.725 | percent}}
{{0.364287465486 | percent}}

currency :

{{0.25 | currency }}
{{0.25 | currency:'GBP'}}
{{ 1.25 | currency: 'EUR'}}
{{ 2345 | currency: 'USD'}}
{{ 947483 | currency: 'INR' : 'code'}}

async pipe:

The output for async pipe: {{ myObservable | async}}

custom pipe:

before sorting: {{ names}}
after sorting: {{ names | psort }}
